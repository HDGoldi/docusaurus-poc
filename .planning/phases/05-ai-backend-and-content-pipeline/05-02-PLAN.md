```typescript
export interface SSEEvent {
  type: 'text' | 'sources' | 'error' | 'done';
  content?: string;
  sources?: Source[];
  error?: string;
}

export interface Source {

  url: string;       // Page URL from metadata
  title: string;     // Page title from metadata
  relevance: number; // Score from Bedrock
}

export interface BedrockCitation {
  generatedResponsePart?: {
    textResponsePart?: {
      text?: string;
      span?: { start: number; end: number };
    };
  };
  retrievedReferences?: Array<{
    content?: { text?: string };
    location?: { s3Location?: { uri?: string } };
    metadata?: Record<string, any>;
  }>;
}
```

**citation-mapper.ts** — per D-09 (inline numbered references):
```typescript
import { BedrockCitation, Source } from './types';

export function mapCitationsToSources(citations: BedrockCitation[]): Source[] {
  const seen = new Map<string, Source>();
  let index = 1;

  for (const citation of citations) {
    for (const ref of citation.retrievedReferences || []) {
      const url = ref.metadata?.url || ref.metadata?.['x-amz-bedrock-kb-source-uri'] || '';
      const title = ref.metadata?.title || 'Untitled';

      if (url && !seen.has(url)) {
        seen.set(url, {
          index: index++,
          url: typeof url === 'string' ? url : String(url),
          title: typeof title === 'string' ? title : String(title),
          relevance: ref.metadata?.score ? Number(ref.metadata.score) : 0,
        });
      }
    }
  }

  return Array.from(seen.values());
}
```

**index.ts** — Lambda streaming handler per D-05 (Claude 3 Haiku), D-06 (eu-central-1), D-08 (SSE), D-10 (honest fallback):
- Use `awslambda.streamifyResponse` for response streaming
- Read `KB_ID` from `process.env.KB_ID`
- Read `MODEL_ARN` from `process.env.MODEL_ARN` with default `arn:aws:bedrock:eu-central-1::foundation-model/anthropic.claude-3-haiku-20240307-v1:0`
- Parse request body for `question` field
- Return 400 SSE error event if question missing or empty
- Call `RetrieveAndGenerateStreamCommand` with:
  - `knowledgeBaseId: process.env.KB_ID`
  - `modelArn: process.env.MODEL_ARN`
  - `maxTokens: 512, temperature: 0.1`
  - `numberOfResults: 5`
- Stream events: write `data: JSON.stringify({ type: 'text', content: event.output.text })\n\n` for each output event
- Collect citations from stream events
- After stream ends, call `mapCitationsToSources(citations)` and write `data: JSON.stringify({ type: 'sources', sources })\n\n`
- Write `data: [DONE]\n\n` as terminal event
- If Bedrock returns no results or an error, write `data: JSON.stringify({ type: 'text', content: "I don't have information about that in the 1NCE documentation. Try asking about connectivity services, SIM management, API authentication, or other topics covered in our developer docs." })\n\n` per D-10
- Set response headers: `Content-Type: text/event-stream`, `Cache-Control: no-cache`, `Connection: keep-alive`
- NOTE: Do NOT set CORS headers in Lambda code. CORS is handled by the Lambda Function URL configuration in CloudFormation (per Research anti-pattern: avoid in-code CORS when using Function URL CORS config). However, since CloudFront proxies the request, CORS headers must come from CloudFront response headers policy instead.
- Wrap entire handler in try/catch — on error, stream an SSE error event and end the stream
